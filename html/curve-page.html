Usage: context.bezierCurveTo(cp1x,cp1y,cp2x,cp2y,x,y);

<script>
var canvas = document.getElementById('myCanvas');
var context = canvas.getContext('2d');
context.beginPath();
context.moveTo(12,112);
context.bezierCurveTo(12, 225, 300, 225, 300, 112);
context.moveTo(12,112);
context.bezierCurveTo(12, 12, 300, 0, 300, 112);
context.moveTo(65,50);

context.moveTo(158, 45);
context.bezierCurveTo(200, 32, 200, 207, 152, 175);
context.lineWidth = 20;
context.strokeStyle = 'black';
